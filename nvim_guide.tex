% --- General ---
\section{General}
\begin{tabular}{@{}ll@{}}
\texttt{<Esc>} & Clear search highlights \\
\texttt{<leader>q} & Open diagnostic [Q]uickfix list \\
\texttt{<leader>f} & [F]ormat buffer (Conform) \\
\texttt{<leader>z} & [Z]en Mode (Toggle Center) \\
\texttt{gcc} / \texttt{gc} & Comment line (Normal / Visual) \\
\texttt{gbc} / \texttt{gb} & Comment block (Normal / Visual) \\
\texttt{u} / \texttt{CTRL-r} & [u]ndo / [r]edo change \\
\texttt{.} & Repeat last edit/mapping \\
\texttt{<Esc><Esc>} & Exit terminal mode (in terminal) \\
\end{tabular}

% --- Basic Vim Motions ---
\section{Vim Motions (Text Navigation)}
\begin{tabular}{@{}ll@{}}
\texttt{h, j, k, l} & Move Left, Down, Up, Right \\
\texttt{w} / \texttt{b} & Move [w]ord fwd / [b]ack \\
\texttt{e} / \texttt{ge} & End of word / End of prev \\
\texttt{0} / \texttt{\$} & Start / End of Line \\
\texttt{\^{}} / \texttt{g\_} & First/Last non-blank char \\
\texttt{\{} / \texttt{\}} & Prev / Next Paragraph \\
\texttt{\%} & Jump to matching pair ()[]\{\} \\
\texttt{gg} / \texttt{G} & Top / Bottom of file \\
\texttt{M, H, L} & [M]id, [H]igh, [L]ow of screen \\
\texttt{gm} & Go to [m]iddle of text line \\
\texttt{*} / \texttt{\#} & Search word under cursor (fwd/bk) \\
\texttt{f/t <char>} & Find/Till next char on line \\
\texttt{F/T <char>} & Find/Till prev char on line \\
\texttt{; / ,} & Repeat last find/till (fwd/back) \\
\end{tabular}
